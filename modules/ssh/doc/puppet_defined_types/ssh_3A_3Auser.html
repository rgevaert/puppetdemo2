define ssh::user (
  $key,
  $comment = '',
  $ensure  = present,
  $user    = undef,
) {

  if $user {
    $username = $user
  } else {
    $username = $name
  }
  
  if $ensure != present {
    Ssh_authorized_key["${name}@${comment}"] -> User[$username]
    Ssh_authorized_key["${name}@${comment}"] -> Group[$username]
  }

  ssh_authorized_key { "${name}@${comment}":
    ensure => $ensure,
    key    => $key,
    user   => $username,
    type   => 'ssh-rsa',
  }
}
